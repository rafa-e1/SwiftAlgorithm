```swift
if let readLine = readLine() {
    let numbers = readLine.split(separator: " ").compactMap { Int($0) }
    if numbers.count == 2 {
        switch (numbers[0], numbers[1]) {
        case let (x, y) where x > y:
            print(">")
        case let (x, y) where x < y:
            print("<")
        case let (x, y) where x == y:
            print("==")
        default:
            break
        }
    } else {
        print("입력 형식이 올바르지 않습니다.")
    }
}
```
