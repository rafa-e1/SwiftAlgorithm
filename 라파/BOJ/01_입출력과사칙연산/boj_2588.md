func multiplyAndPrint(_ inputNumber1: Int, _ inputNumber2: Int) {
    let a = inputNumber1
    let b = inputNumber2
    
    let b1 = b / 100
    let b2 = (b / 10) % 10
    let b3 = b % 10
    
    let result1 = a * b3
    let result2 = a * b2
    let result3 = a * b1
    let finalResult = a * b
    
    print(result1)
    print(result2)
    print(result3)
    print(finalResult)
}

let inputNumber1 = Int(readLine()!)!
let inputNumber2 = Int(readLine()!)!

multiplyAndPrint(inputNumber1, inputNumber2)
