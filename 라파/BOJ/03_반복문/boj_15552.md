```swift
struct TestCase {
    let a: Int
    let b: Int
}

func readInput() -> [TestCase] {
    guard let t = Int(readLine() ?? "") else { return [] }
    var testCases = [TestCase]()
    
    for _ in 0..<t {
        let numbers = (readLine() ?? "").split(separator: " ").compactMap { Int($0) }
        if numbers.count == 2 {
            testCases.append(TestCase(a: numbers[0], b: numbers[1]))
        }
    }
    
    return testCases
}

func processTestCases(_ testCases: [TestCase]) -> [Int] {
    return testCases.map { $0.a + $0.b }
}

func main() {
    let testCases = readInput()
    let results = processTestCases(testCases)
    results.forEach { print($0) }
}

main()
```
